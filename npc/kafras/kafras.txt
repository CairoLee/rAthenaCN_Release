aldeba_in,96,181,4	script	卡普拉服务	113,{
	cutin "kafra_05",2;
	mes "[卡普拉 克拉丽斯]";
	mes "嗯...?";
	mes "噢, 欢迎来到卡普拉公司总部.";
	mes "您需要什么服务呢?";
	next;
	switch(select("储存服务:仓库服务:租用手推车:结束对话")) {
	case 1:
		mes "[卡普拉 克拉丽斯]";
		mes "您的复活点已经存储到了:";
		mes "^0000FF卡普拉公司总部^000000";
		mes "谢谢您使用我们卡普拉公司的服务!";
		next;
		savepoint "aldeba_in",96,179,1,1;
		mes "[卡普拉 克拉丽斯]";
		mes "您可以在卢恩・米德加尔特王国中随意";
		mes "使用我们卡普拉服务!";
		mes "感谢您光临我们卡普拉总部.";
		close2;
		break;
	case 2:
		if (Zeny < 20) {
			mes "[卡普拉 克拉丽斯]";
			mes "很抱歉!";
			mes "您看起来好像没有 20 Zeny 可用来";
			mes "可以支付仓库的开启费用...";
			close;
		}
		mes "[卡普拉 克拉丽斯]";
		mes "虽然这个设施是卡普拉的管理培训机构,";
		mes "但是我们依然会帮您开启仓库.";
		next;
		set Zeny, Zeny-20;
		set RESRVPTS, RESRVPTS + (20/5);
		mes "[卡普拉 克拉丽斯]";
		mes "以后如果想使用卡普拉服务,";
		mes "请随时寻找您身边的卡普拉职员,";
		mes "她们会竭诚为您服务. 感谢您的惠顾!";
		callfunc("F_CheckKafCode","[卡普拉 克拉丽斯]");    //check your storage password, if set
		close2;
		openstorage;
		break;
	case 3:
		mes "[卡普拉 克拉丽斯]";
		mes "很抱歉, 目前我没有在值班,";
		mes "没有支取手推车的权限,";
		mes "无法为您提供手推车租用服务~";
		mes "建议您去问问其他值班中的卡普拉职员~";
		next;
		mes "[卡普拉 克拉丽斯]";
		mes "卡普拉职员几乎覆盖了整个卢恩・米德加尔特王国.";
		mes "您可以在各个城镇甚至野外找到尽心为您服务的卡普拉职员.";
		close2;
		break;
	case 4:
		mes "[卡普拉 克拉丽斯]";
		mes "^666666*呼...*^000000";
		mes "太好了, 实际上我正在休息~";
		mes "结束对话是一个不错的选择, 嘻嘻...";
		close2;
		break;
	}
	cutin "",255;
	end;
}
